// Cron칩metro
let tiempo=0, intervalo=null;
const cronometro=document.getElementById("cronometro");
function iniciarCronometro() {
  if(!intervalo){
    intervalo=setInterval(()=>{
      tiempo++;
      const min=Math.floor(tiempo/60);
      const sec=tiempo%60;
      cronometro.innerText=`Tiempo: ${String(min).padStart(2,"0")}:${String(sec).padStart(2,"0")}`;
    },1000);
  }

const sonidoAcierto=document.getElementById("sonidoAcierto");

  div.classList.add("carta");
  const frente=document.createElement("div");

  frente.innerText="";
  const atras=document.createElement("div");

  atras.innerText=texto;
  div.appendChild(frente);

  div.dataset.valor=texto;
  div.addEventListener("click",()=>voltear(div));
  tablero.appendChild(div);

// Funciones del juego
function voltear(carta){
  iniciarCronometro();
  if(bloqueo || carta.classList.contains("volteada") || carta.classList.contains("acertada")) return;
  carta.classList.add("volteada");
  if(!primera){ 
    primera=carta; 
  } else{
    intentos++; 
    document.getElementById("intentos").innerText="Intentos: "+intentos;
    if(sonPareja(primera.dataset.valor,carta.dataset.valor)){
      primera.classList.add("acertada"); 
      carta.classList.add("acertada"); 
      encontrados++; 
      sonidoAcierto.play();
      if(encontrados===pares.length){ 
        clearInterval(intervalo); 
        document.getElementById("mensaje").innerText=`游꿀 춰Felicidades! Lo lograste en ${intentos} intentos y ${String(Math.floor(tiempo/60)).padStart(2,"0")}:${String(tiempo%60).padStart(2,"0")}.`; 
        sonidoVictoria.play(); 
      }
      primera=null;
    } else{
      bloqueo=true; 
      sonidoError.play();
      setTimeout(()=>{ 
        primera.classList.remove("volteada"); 
        carta.classList.remove("volteada"); 
        primera=null; 
        bloqueo=false; 
      },1000);
    }
  }
}
function sonPareja(a,b){ 
  return pares.some(par=>par.includes(a)&&par.includes(b)&&a!==b); 
}
